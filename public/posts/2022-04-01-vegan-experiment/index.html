Day 1: Secret Entrance

May 26, 2026

package main

import (
"bufio"
"fmt"
"os"
"strconv"
)

type Simulation struct {
Moves []int
InitialPosition int
TotalPositions int
Target int
}

func main() {
moves, err := ParseInstructionsFromFile("input.txt")
if err != nil {
fmt.Println("Failed to parse instructions: ", err)
return
}

    sim, err := NewSimulation(moves, 50, 100, 0)
    if err != nil {
    	fmt.Println("Simulation could not be created:", err)
    	return
    }

    result, err := sim.Run()

    if err != nil {
    	fmt.Println("Simulation failed: ", err)
    	return
    }

    fmt.Println(sim.Target, "was hit", result, "times.")

}

func NewSimulation(moves []int, start, total, target int) (\*Simulation, error) {
if total <= 0 {
return nil, fmt.Errorf("Total positions must be greater than 0")
}

    return &Simulation{
    	Moves:           moves,
    	InitialPosition: start,
    	TotalPositions:  total,
    	Target:          target,
    }, nil

}

func ParseInstructionsFromFile(filename string) (movesList []int, err error) {
file, err := os.Open(filename)
if err != nil {
return nil, fmt.Errorf("Failed to open file: %v", err)
}
defer file.Close()

    movesList = []int{}
    scanner := bufio.NewScanner(file)

    for scanner.Scan() {
    	line := scanner.Text()

    	if len(line) == 0 {
    		continue
    	}

    	distance, err := strconv.Atoi(line[1:])
    	if err != nil {
    		return nil, fmt.Errorf("line %s: %w", line, err)
    	}

    	switch line[0] {
    	case 'R':
    		movesList = append(movesList, distance)
    	case 'L':
    		movesList = append(movesList, -distance)
    	default:
    		return nil, fmt.Errorf("invalid direction '%c' on line: %s", line[0], line)
    	}
    }

    if err := scanner.Err(); err != nil {
    	return nil, fmt.Errorf("scan: %w", err)
    }

    return movesList, nil

}

func (s \*Simulation) Run() (int, error) {

    count := 0
    currentPosition := s.InitialPosition

    for _, move := range s.Moves {
    	position := currentPosition + move
    	currentPosition = s.Wrap(position)

    	if currentPosition == s.Target {
    		count++
    	}
    }
    return count, nil

}

func (s \*Simulation) Wrap(position int) int {
return ((position % s.TotalPositions) + s.TotalPositions) % s.TotalPositions
}
