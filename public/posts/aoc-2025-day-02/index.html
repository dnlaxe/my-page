Part 1

The Gist

Given a list of comma-separated number ranges, scan for invalid IDs formed by a sequence of digits repeated exactly twice (such as 55 or 123123) with no leading zeroes, and calculate the total sum of these numbers.

Slicing numbers without string manipulation

Using pure math is better than string manipulation because…

Math happens in the stack, string manipulation needs memory on the heap.

After Math happens it vanishes, dead strings need to be garbage collected.

Integers are easier for CPU to process than strings which are arrays of bytes.

My final code

package main

import (
	"fmt"
	"log"
	"math"
	"os"
	"strconv"
	"strings"
)

type Range struct {
	Start int
	End   int
}

func main() {

	ranges, err := ParseNumbersFromFile("input.txt")
	if err != nil {
		log.Fatalf("Fatal: Failed to read input file: %v\n", err)
	}

	invalidNumbersSum := 0

	for _, r := range ranges {
		for i := r.Start; i <= r.End; i++ {
			if hasRepeatingHalves(i) {
				invalidNumbersSum += i
			}
		}
	}
	fmt.Println(invalidNumbersSum)
}

func ParseNumbersFromFile(filename string) (ranges []Range, err error) {
	data, err := os.ReadFile(filename)
	if err != nil {
		return nil, fmt.Errorf("Failed to open file: %v", err)
	}

	fileString := string(data)

	stringPairs := strings.Split(fileString, ",")

	var finalPairs []Range

	for _, pair := range stringPairs {

		stringNums := strings.Split(pair, "-")

		if len(stringNums) != 2 {
			log.Printf("Warning: Skipping malformed pair (missing hyphen) '%s'\n", pair)
			continue
		}

		first, err1 := strconv.Atoi(stringNums[0])
		second, err2 := strconv.Atoi(stringNums[1])

		if err1 != nil || err2 != nil {
			log.Printf("Warning: Skipping invalid line '%s'\n", pair)
			continue
		}

		parsedPair := NewRange(first, second)

		finalPairs = append(finalPairs, parsedPair)
	}

	return finalPairs, nil
}

func hasRepeatingHalves(number int) bool {
	numberLength := len(strconv.Itoa(number))
	if numberLength%2 != 0 {
		return false
	}

	halfLength := numberLength / 2
	divisor := int(math.Pow(10, float64(halfLength)))
	first := number / divisor
	second := number % divisor

	return (first == second)
}

func NewRange(start, end int) Range {
	if start > end {
		return Range{
			Start: end,
			End:   start,
		}
	}
	return Range{
		Start: start,
		End:   end,
	}
}

Part 2

The Gist

Given a list of comma-separated number ranges, scan for invalid IDs formed by a sequence of digits repeated at least twice (such as 12341234 or 111) with no leading zeroes, and calculate the total sum of these numbers.

Splitting integers

Integers need to be converted to strings before they can split. Splitting numbers can be done with math but it si ineffective as any part beginning with a zero will have it automatically omitted, for example [100100] splitting into three parts would return [10 1 0], not [10 01 00]

func findFactors(number int) ([]int, error) {
	var factors []int
	start := 1
	end := number / 2
	for i := start; i <= end; i++ {
		if number%i == 0 {
			factors = append(factors, i)
		}
	}
	return factors, nil
}

func splitIntoParts(number int, length int) ([]string, error) {
	text := strconv.Itoa(number)

	var result []string
	for i := 0; i < len(text); i += length {
		end := i + length
		if end > len(text) {
			end = len(text)
		}
		result = append(result, text[i:end])
	}
	return result, nil
}

func hasRepeatingParts(parts []string) bool {
	chunk := parts[0]

	for _, val := range parts[1:] {
		if val != chunk {
			return false
		}
	}
	return true
}
