# Part 1

## The Gist

Given a list of battery banks represented as strings of digits, identify the two digits within each bank that form the highest possible numerical value while maintaining their original relative order, then calculate the sum of these maximum joltages across all banks.

## Code for part 1

```
package main

import (
	"bufio"
	"fmt"
	"os"
)

type BatteryRow struct {
	Readings []int
}

func main() {
	data, err := ParseDataFromFile("input.txt")
	if err != nil {
		fmt.Println(err)
	}

	answer := run(data)

	fmt.Println(answer)
}

func ParseDataFromFile(filename string) ([]BatteryRow, error) {

	file, err := os.Open(filename)
	if err != nil {
		return nil, fmt.Errorf("Failed to open file: %v", err)
	}
	defer file.Close()

	var batteryList []BatteryRow

	scanner := bufio.NewScanner(file)

	for scanner.Scan() {
		line := scanner.Text()

		row := BatteryRow{
			Readings: make([]int, 0, len(line)),
		}

		for _, letter := range line {
			row.Readings = append(row.Readings, int(letter-'0'))
		}

		batteryList = append(batteryList, row)
	}

	if err := scanner.Err(); err != nil {
		return nil, fmt.Errorf("Scanning error")
	}

	return batteryList, nil
}

func run(data []BatteryRow) int {
	sum := 0
	var numbers []int
	for _, row := range data {
		length := len(row.Readings)
		tens := 0
		ones := 0
		for index, num := range row.Readings {
			if num > tens && index != (length-1) {
				tens = num
				ones = 0
				continue

			}
			if num > ones {
				ones = num
				continue
			}
		}
		numbers = append(numbers, (tens*10)+ones)
	}

	for _, val := range numbers {
		sum += val
	}
	return sum
}

```

# Part 2

## The Gist

Given the same list of battery banks, identify the twelve digits within each bank that form the highest possible 12-digit numerical value while maintaining their original relative order, then calculate the total sum of these larger maximum joltages across all banks.

## The Process

Create a slice of zeros to store answers in. Take each number from the battery line one by one. Starting at index zero compare them with the numbers in the slice. If it is bigger, replace the current number with that number and then change every following number in the slice to zero.

A constraint is needed to ensure that the slice ends with no zero values remaining. The condition `attempt.NumbersRemaining >= attempt.SlotsLength-index-1` ensures that numbers are at the end of battery lines cannot replace numbers in the slice that are too far away. If they could replace these values, zeroes would be left in the final answer.

## Code for part 2

```
func run(data []BatteryRow) int {
	sum := 0
	for _, row := range data {
		sum += getAnswer(row, 12)
	}
	return sum
}

type Process struct {
	BatteryRow       []int
	SlotsRow         []int
	NumbersRemaining int
	SlotsRemaining   int
	SlotsLength      int
}

func getAnswer(row BatteryRow, slotLength int) int {

	initialSlots := make([]int, slotLength)

	attempt := Process{
		BatteryRow:       row.Readings,
		SlotsRow:         initialSlots,
		NumbersRemaining: len(row.Readings),
		SlotsRemaining:   len(initialSlots),
		SlotsLength:      len(initialSlots),
	}

	for _, digit := range attempt.BatteryRow {
		attempt.NumbersRemaining--
		replaced := false

		for index, slot := range attempt.SlotsRow {

			if replaced {
				attempt.SlotsRow[index] = 0
			}

			if digit > slot && attempt.NumbersRemaining >= attempt.SlotsLength-index-1 && !replaced {
				attempt.SlotsRow[index] = digit
				attempt.SlotsRemaining--
				replaced = true
			}
		}
	}
	var answer int = 0
	for _, n := range attempt.SlotsRow {
		answer = (answer * 10) + int(n)
	}
	return answer
}
```
